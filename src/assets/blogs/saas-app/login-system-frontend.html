üîÑ Handling Authentication in Vue

The real magic happens in our script section:

javascript
 
import { ref } from 'vue'
import router from '../router'
import axios from 'axios'

// Reactive state
const activeTab = ref('login')
const loginForm = ref({ email: '', password: '' })
const registerForm = ref({ name: '', email: '', password: '' })
const loginError = ref(null)
const registerError = ref(null)

// Login handler
const handleLogin = async () => {
  try {
    const res = await axios.post('/api/login', {
      email: loginForm.value.email,
      password: loginForm.value.password
    })

    localStorage.setItem('token', res.data.token)
    router.push({ name: 'dashboard' })
  } catch (err) {
    loginError.value = err.response?.data?.message || 'Login failed'
  }
}

// Registration handler
const handleRegister = async () => {
  try {
    const res = await axios.post('/api/register', registerForm.value)
    localStorage.setItem('token', res.data.token)
    router.push({ name: 'dashboard' })
  } catch (err) {
    registerError.value = err.response?.data?.message || 'Registration failed'
  }
}

Security Considerations:

  • Tokens stored in localStorage (consider httpOnly cookies for production)

  • Error messages don't reveal system information

  • Password never logged or stored improperly

  • HTTPS required for all requests
